---
title: "Get bot feedback"
api: "GET https://simo-botlist.vercel.app/api/bots/{botID}/feedbacks"
description: "This method is used to get all feedback already made to a bot."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The api key of a bot that you own. You can find in [dashboard](https://bombadeagua.life/dashboard/) after selecting a bot.
</ParamField>
<ParamField type="string" path="botID" required>
  The bot id
</ParamField>

### Response

<ResponseField name="Feedback Structure" type="object">
    The current bot feedbacks
    <Expandable>
        <ResponseField name="stars" type="number">
            The feedback stars
        </ResponseField>
        <ResponseField name="content" type="string">
            The feedback content
        </ResponseField>
        <ResponseField name="target_bot_id" type="string">
            The feedback target bot id
        </ResponseField>
        <ResponseField name="edited" type="boolean">
            Indicates whether the feedback has been edited
        </ResponseField>
        <ResponseField name="reply_message" type="object">
            Information about the reply message
            <Expandable>
                <ResponseField name="content" type="string">
                    The content of the reply message
                </ResponseField>
                <ResponseField name="posted_at" type="string">
                    The timestamp when the reply message was posted
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="author" type="object">
            Information about the author of the feedback
            <Expandable>
                <ResponseField name="username" type="string">
                    The username of the author
                </ResponseField>
                <ResponseField name="avatar" type="string">
                    The avatar of the author
                </ResponseField>
                <ResponseField name="id" type="string">
                    The unique identifier of the author
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="posted_at" type="string">
            The timestamp when the feedback was originally posted
        </ResponseField>
    </Expandable>
</ResponseField>
