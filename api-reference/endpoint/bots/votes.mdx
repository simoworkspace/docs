---
title: "Get bot votes"
api: "GET https://simo-botlist.vercel.app/api/bots/{botID}/votes"
description: "This method is used to get all votes from a bot."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The api key of a bot that you own. You can find in [dashboard](https://bombadeagua.life/dashboard/) after selecting a bot.
</ParamField>
<ParamField type="string" path="botID" required>
  The bot id
</ParamField>

<ResponseField name="votes" type="object">

    <ResponseField name="user_id" type="string">

    <ResponseField name="last_vote" type="string">
        User last vote
    </ResponseField>
     <ResponseField name="votes" type="number">
        User votes
    </ResponseField>
    </Expandable> 
</ResponseField>
