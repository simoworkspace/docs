<ResponseField name="API Status Structure" type="object">

    </ResponseField>
    <ResponseField name="users" type="number">

    </ResponseField>
    <ResponseField name="bots" type="number">
      Total added bots

      API uptime
    </ResponseField>

  </Expandable>    
</ResponseField>
