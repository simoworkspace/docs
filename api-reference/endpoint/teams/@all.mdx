---
title: "Get user's teams"
api: "GET https://simo-botlist.vercel.app/api/teams/@all"
description: "Search for all teams where the user is a member."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The api key of a bot that you own. You can find in [dashboard](https://bombadeagua.life/dashboard/) after selecting a bot.
</ParamField>

### Response

<ResponseField name="Team" type="object">
    Array of the information about the team
    <Expandable>
        <ResponseField name="bots_id" type="array">
            The ID(s) of bots in the team
        </ResponseField>
        <ResponseField name="description" type="string">
            The description of the team
        </ResponseField>
        <ResponseField name="avatar_url" type="string">
            The URL of the team's avatar
        </ResponseField>
        <ResponseField name="name" type="string">
            The name of the team
        </ResponseField>
        <ResponseField name="invite_code" type="string">
            The invite code for the team
        </ResponseField>
        <ResponseField name="id" type="string">
            The unique identifier of the team
        </ResponseField>
        <ResponseField name="members" type="array">
            Information about team members
            <Expandable>
                <ResponseField name="joined_at" type="string">
                    The timestamp when the member joined the team
                </ResponseField>
                <ResponseField name="id" type="string">
                    The unique identifier of the member
                </ResponseField>
                <ResponseField name="permission" type="number">
                    The permission level of the member
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="created_at" type="string">
            The timestamp when the team was created
        </ResponseField>
    </Expandable>
</ResponseField>
