---
title: "Get team members"
api: "GET https://simo-botlist.vercel.app/api/teams/{teamID}/members"
description: "Search for all members of a team."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The api key of a bot that you own. You can find in [dashboard](https://bombadeagua.life/dashboard/) after selecting a bot.
</ParamField>
<ParamField type="string" path="teamID" required>
  The team id
</ParamField>

### Response

<ResponseField name="TeamMember" type="object">
    Array of team members
    <Expandable>
        <ResponseField name="id" type="string">
            The ID of the member
        </ResponseField>
        <ResponseField name="permission" type="number">
            The permission level of the member in the team
        </ResponseField>
        <ResponseField name="joined_at" type="string">
            An ISO string representing when the member joined the team
        </ResponseField>
    </Expandable>
</ResponseField>
