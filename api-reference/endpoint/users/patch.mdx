---
title: "Edit user"
api: "PATCH https://simo-botlist.vercel.app/api/users"
description: "Update the user object that matches the JWT or api-key."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The JWT of your account. You can get this in website cookies
</ParamField>
<ParamField path="bio" type="string" maxLength="200">
    The user biography (1-200 characters)
</ParamField>
<ParamField path="notifications_viewed" type="boolean">
    Indicates whether the user has viewed notifications
</ParamField>
<ParamField path="banner_url" type="string">
    The URL of the user's banner
</ParamField>

- The user ID will be taken from the JWT or api key used

### Response

<ResponseField name="User" type="object">
    Returns the updated user object
    <Expandable>
        <ResponseField name="_id" type="string">
            The user id
        </ResponseField>
        <ResponseField name="username" type="string">
            The user username
        </ResponseField>
        <ResponseField name="avatar" type="string">
            The user avatar
        </ResponseField>
        <ResponseField name="notifications" type="object">
            The user notifications
            <Expandable>
                <ResponseField name="content" type="string">
                    The content of the notification
                </ResponseField>
                <ResponseField name="sent_at" type="string">
                    An ISO string representing when the notification was created
                </ResponseField>
                <ResponseField name="type" type="number">
                    The type of the notification
                </ResponseField>
                <ResponseField name="url" type="string" optional="true">
                    The hyperlink of the notification (Available only with Comment type)
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="bio" type="string">
            The user biography
        </ResponseField>
        <ResponseField name="notifications_viewed" type="boolean">
            Indicates whether the user has viewed notifications
        </ResponseField>
    </Expandable>
</ResponseField>
